weekly_users = profiles_filtered.groupby(['week', channel_col]).size().reset_index()
weekly_users.columns = ['week', 'channel', 'new_users']

]

        print(f"\n{'='*60}")
        print(f"АНАЛИЗ КАНАЛА: {channel}")
        print('='*60)
        
        # Тепловая карта
        plot_cohort_roi_heatmap(channel)

            avg_roi = channel_data['roi'].mean()
            avg_cac = channel_data['cac'].mean()
            avg_ltv_10w = channel_data.groupby('first_week')['ltv'].sum().mean()
            
            # Анализ окупаемости
            weekly_roi = channel_data.groupby('N_week')['roi'].mean().reset_index()
            payback_week = weekly_roi[weekly_roi['roi'] >= 0]['N_week'].min()
            
            if pd.notna(payback_week):

            })

                print(f"  Средний ROI: {row['Средний ROI']:.2f}, Рекомендация: {row['Рекомендация']}")
            print()
    else:
        print("Нет данных для анализа проблемных каналов")
